 15 	/**
 16 	* Returns whether this object can respond to the _methodName parameter.
 17 	* @param {String} _methodName
 18 	* @return {Boolean}
 19 	*/
 20 	Object.prototype.respondsTo = function(_methodName)
 21 	{
 22 		return this.getClass().getMethod(_methodName) != null;
 23 	}

 25 	/**
 26 	* Returns whether the _object parameter is 'undefined'.
 27 	* @param {Object} _object
 28 	* @return {Boolean}
 29 	*/
 30 	Object.isUndefined = function(_object)
 31 	{
 32 		return _object == Object.NOT_DEF;
 33 	}

 35 	/**
 36 	* Returns whether this object is equal to the specifiec _object.
 37 	* @param {Object} _object
 38 	* @return {Boolean}
 39 	*/
 40 	Object.prototype.equals = function(_object)
 41 	{
 42 		if(this === _object)
 43 		{
 44 			return true;
 45 		}
 46 		if(!this.getClass().isInstance(_object))
 47 		{
 48 			return false;
 49 		}
 50 
 51 		return this == _object;
 52 	}

 54 	/**
 55 	* Returns the default representation of the object.
 56 	* @return {String}
 57 	*/
 58 	Object.prototype.toString = function()
 59 	{
 60 		return this.getClass().getName();
 61 	}
