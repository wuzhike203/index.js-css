 49 	jsx.dom.behavior.draggable.Draggable.mousedown_mixin = function(_nativeEvent)
 50 	{
 51 		this.mousedown(_nativeEvent);
 52 
 53 		var _style = this.container.style;
 54 		if(!_style.left || !_style.top)
 55 		{
 56 			var _pos = jsx.dom.Element.getPosition(this.container);
 57 			_style.position = "absolute"
 58 			_style.left = _pos.left + "px";
 59 			_style.top = _pos.top + "px";
 60 
 61 			jsx.dom.behavior.draggable.Draggable.Z_INDEX += 1;
 62 
 63 			_style.zIndex = jsx.dom.behavior.draggable.Draggable.Z_INDEX;
 64 		}
 65 
 66 		var _event = new jsx.event.Event(_nativeEvent);
 67 
 68 		this.container.diff_x = _event.getPointerX() - Number.parseInt(_style.left);
 69 		this.container.diff_y = _event.getPointerY() - Number.parseInt(_style.top);
 70 
 71 		_event.stopDefault();
 72 		_event.stopSelecting();
 73 	}

 85 	{
