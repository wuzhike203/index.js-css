 16 	jsx.event.EventDispatcher.prototype.initialize = function(_object, _method, _useDomEvent)
 17 	{
 18 		this.object = _object;
 19 		this.method = _method;
 20 		this.useDomEvent = _useDomEvent;
 21 		this.listeners = new jsx.collections.ArrayList();
 22 
 23 		if(this.useDomEvent)
 24 		{
 25 			if(this.method.indexOf("on") == -1)
 26 			{
 27 				this.method = "on"+this.method;
 28 			}
 29 		}
 30 
 31 		// if the object already has something set for the event, add it to the listeners
 32 		var _listener = this.object[this.method];
 33 		if(_listener)
 34 		{
 35 			this.addListener(_listener);
 36 		}
 37 
 38 		if(this.useDomEvent)
 39 		{
 40 			this.eadd = this.object.addEventListener ? "addEventListener" : "attachEvent";
 41 			this.ermv = this.object.removeEventListener ? "removeEventListener" : "detachEvent";
 42 			this.ename = this.method;
 43 			if(this.object.addEventListener) this.ename = this.method.replace("on", "");
 44 		}
 45 		else
 46 		{
 47 			// set the event method to reference the event dispatcher dispatch method
 48 			this.object[this.method] = jsx.event.EventDispatcher.dispatch;
 49 
 50 			// create a property on the object|element pointing to "this" event dispatcher
 51 			this.object[this.method+"_event"] = this;
 52 		}
 53 
 54 		// used to cache objects|elements
 55 		jsx.event.EventDispatcher.cache.add([this.object, this.method]);
 56 	}

 58 	jsx.event.EventDispatcher.prototype.addListener = function(_listener, _capture)
 59 	{
 60 		if(!_listener) return;
 61 
 62 		if(this.useDomEvent) this.object[this.eadd](this.ename, _listener, (_capture?_capture:false));
 63 
 64 		this.listeners.add(_listener);
 65 		return _listener;
 66 	}

 68 	jsx.event.EventDispatcher.prototype.removeListener = function(_listener, _capture)
 69 	{
 70 		if(!_listener) return;
 71 
 72 		if(this.useDomEvent) this.object[this.ermv](this.ename, _listener, (_capture?_capture:false));
 73 
 74 		this.listeners.remove(_listener);
 75 		return _listener;
 76 	}

 79 	{
 80 		var me = this;
 81 		this.listeners.each(function(_function)
 82 		{
 83 			me.removeListener(_function, false);
 84 			me.removeListener(_function, true);
 85 		});

 88 	jsx.event.EventDispatcher.prototype.notify = function(_event)
 89 	{
 90 		var me = this;
 91 		this.listeners.each(function(_function)
 92 		{
 93 			_function.apply(me.object, [_event]);
 94 		});
 95 	}

 97 	// when called by the client, need to pass the _event object and set the target and type properties in the event,
 98 	// only need to explictly create a new _event object if not using a DOM Element
 99 	jsx.event.EventDispatcher.dispatch = function(_event)
100 	{
101 		// the event object could be passed in, if not use the global object event
102 		var _event = _event || $GLOBAL_OBJECT.event;
103 
104 		// source object of the event
105 		var _source = _event.target || _event.srcElement;
106 
107 		// event type, really the method
108 		var _type = "on"+_event.type+"_event";
109 
110 		if(_source && _source[_type]) _source[_type].notify(_event);
111 		else if($GLOBAL_OBJECT[_type]) $GLOBAL_OBJECT[_type].notify(_event);
112 		else if(_event.currentTarget && _event.currentTarget[_type]) _event.currentTarget[_type].notify(_event);
113 	}
