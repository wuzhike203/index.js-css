336 	jsx.io.http.HttpRequest.prototype.handleStateChange = function(_state)
337 	{
338         if(this.canceled) return;
339 		try
340 		{
341 			switch(_state)
342 			{
343 				case jsx.io.http.HttpRequest.SENT:
344 					// fire onsent event
345 					this.onsent({target:this,type:"sent"});
346 					break;
347 				case jsx.io.http.HttpRequest.RECEIVING:
348 					// fire onreceiving event
349 					this.onreceiving({target:this,type:"receiving"});
350 					break;
351 				case jsx.io.http.HttpRequest.RECEIVED:
352 					// if we have an abort watching, since we just completed, cancel the watching
353 					if(this.cancelThread)
354 					{
355 						this.cancelThread.stop();
356 					}
357 
358 					// set the response
359 					this.response = new jsx.io.http.HttpResponse(this.transport);
360 
361 					// fire onreceived event
362 					this.onreceived({target:this,type:"received"});
363 
364 					if(this.response.isSuccess())
365 					{
366 						// fire onsuccess event
367 						this.onsuccess({target:this,type:"success"});
368 					}
369 					else
370 					{
371 						// fire onfailure event
372 						this.onfailure({target:this,type:"failure"});
373 					}
374 
375 					// fire oncomplete event
376 					this.oncomplete({target:this,type:"complete"});
377 
378 					this.complete = true;
379 
380 					// clean up closure
381 					this.transport.onreadystatechange = function(){};
382 					break;
383 				default:
384 					throw "State: " + _state + " was not handled!";
385 			}
386 		}

390 		}
