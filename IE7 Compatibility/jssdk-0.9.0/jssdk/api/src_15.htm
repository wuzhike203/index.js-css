 18 	* @param {Boolean} _reversal
 19 	*/
 20 	jsx.collections.comparators.ComparatorUtil.compare = function(_objectA, _objectB, _reversal)
 21 	{
 22 		var a = _objectA;
 23 		var b = _objectB;
 24 
 25 		var comp;
 26 
 27 		if(a < b) comp = -1;
 28 		if(a == b) comp = 0;
 29 		if(a > b) comp = 1;
 30 
 31 		if(_reversal)
 32 		{
 33 			return -1 * comp;
 34 		}
 35 
 36 		return comp;
 37 	}
