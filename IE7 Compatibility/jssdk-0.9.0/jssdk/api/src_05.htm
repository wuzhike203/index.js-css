 46 	* @param {Number} _index
 47 	* @param {Object} _object
 48 	*/
 49 	jsx.collections.ArrayList.prototype.addAt = function(_index, _object)
 50 	{
 51 		if(_index < 0){return false;}

 54 		return true;

 59 	* Removes and returns the element at the specified position in this list.
 60 	* @param {Number} _index
 61 	* @return {Object}
 62 	*/
 63 	jsx.collections.ArrayList.prototype.removeAt = function(_index)
 64 	{
 65 		if(_index < 0){ return false; }
 66 
 67 		var items = this.entryItems.splice(_index, 1);
 68 
 69 		if(items.length == 0){ return false; }
 70 
 71 		return true;
 72 	}

 95 	/**
 96 	* Returns a view of the portion of this list between fromIndex, inclusive, and toIndex, exclusive.
 97 	* @param {Number} _fromIndex
 98 	* @param {Number} _toIndex
 99 	* @reutrn {jsx.collections.List}
100 	*/
101 	jsx.collections.ArrayList.prototype.subList = function(_fromIndex, _toIndex)
102 	{
103 		var sublist = new jsx.collections.ArrayList();
104 
105 		this.inject(0, function(_memo, _value, _index)
106 		{
107 			if(_index >= _fromIndex && _index <= _toIndex)
108 			{
109 				sublist.add(_value);
110 			}
111 
112 			return _memo;
113 		});
114 
115 		return sublist;
116 	}
