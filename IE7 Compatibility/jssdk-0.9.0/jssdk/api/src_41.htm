  6 	/**
  7 	* @class Closure
  8 	* @constructor
  9 	* @description A Closure is a function that is evaluated in an environment containing one or more bound variables.
 10 	*  When called, the function can access these variables. In this case the _block parameter in the bind method will have
 11 	*  access to variable in the same scope as where the closure is defined.
 12 	*/
 13 	jsx.lang.Closure = function(){this.initialize.apply(this, arguments);}

 18 	/**
 19 	* Returns a Function. The _object parameter is appended to the _args parameter when the _block is excuted. If the
 20 	*  Closure is bound to an event, the event is appended to the _args parameter when the _block is executed.
 21 	* @param {Function} _block Block to execute
 22 	* @param {Object} _object Context Object
 23 	* @param {Array} [_args] Array of parameters to pass into the block
 24 	* @return {Function}
 25 	*/
 26 	jsx.lang.Closure.prototype.bind = function(_block, _object, _args)
 27 	{
 28 		var _args = _args || [];
 29 
 30 		return function(_event) {
 31 
 32 			_args = _args.concat([_object]);
 33 
 34 			// both _event or global object event could be null, which means the result could be null if this closure
 35 			// is not bound to an event
 36 			var _event = _event || $GLOBAL_OBJECT.event;
 37 			if(_event)
 38 			{
 39 				_args = _args.concat([_event]);
 40 
 41 			}
 42 
 43 			return _block.apply(_object, _args);
 44 		}
 45 	}
