 22 	jsx.logging.PatternLayout.prototype.format = function(_logEvent)
 23 	{
 24 		var _msg = new jsx.lang.StringBuffer();
 25 
 26 		var _regexp = new RegExp(jsx.logging.PatternLayout.__PATTERN_REG_EXP__, "g");
 27 		var _result;
 28 		while((_result = _regexp.exec(this.pattern)) != null)
 29 		{
 30 			var _match = _result[0];
 31 			var _padding = _result[1];
 32 			var _truncation = _result[2];
 33 			var _conversionCharacter = _result[3];
 34 			//_result[4] // specifier with "{" and "}"
 35 			var _specifier = _result[5];
 36 			var _text = _result[6];
 37 
 38 			var _replacement = "";
 39 			if(_text)
 40 			{
 41 				_msg.append(_text)
 42 				continue;
 43 			}
 44 
 45 			switch(_conversionCharacter)
 46 			{
 47 					case "c": // Logger name
 48 						var _loggerName = _logEvent.getLogger().getName();
 49 						if(_specifier)
 50 						{
 51 							var _precision = Number.parseInt(_specifier);
 52 							var _loggerNameParts = _logEvent.getLogger().getName().split(".");
 53 							if(_precision >= _loggerNameParts.length)
 54 							{
 55 								_replacement = _loggerName;
 56 							}
 57 							else
 58 							{
 59 								_replacement = _loggerNameParts.slice(_loggerNameParts.length - _precision).join(".");
 60 							}
 61 						}
 62 						else
 63 						{
 64 							_replacement = _loggerName;
 65 						}
 66 						break;
 67 					case "d": // Date
 68 						var _dateFormat = jsx.logging.PatternLayout.ISO8601_DATEFORMAT;
 69 						if(_specifier)
 70 						{
 71 							_dateFormat = _specifier;
 72 							// Pick up special cases
 73 							if(_dateFormat == "ISO8601")
 74 							{
 75 								_dateFormat = jsx.logging.PatternLayout.ISO8601_DATEFORMAT;
 76 							}
 77 							else if(_dateFormat == "ABSOLUTE")
 78 							{
 79 								_dateFormat = jsx.logging.PatternLayout.ABSOLUTETIME_DATEFORMAT;
 80 							}
 81 							else if(_dateFormat == "DATE")
 82 							{
 83 								_dateFormat = jsx.logging.PatternLayout.DATETIME_DATEFORMAT;
 84 							}
 85 						}
 86 						_replacement = new jsx.lang.util.SimpleDateFormat(_dateFormat).format(_logEvent.getTimeStamp());
 87 						break;
 88 					case "m": // Message
 89 						_replacement = _logEvent.getMessage().toString();
 90 						break;
 91 					case "n": // New line
 92 						_replacement = "\n";
 93 						break;
 94 					case "p": // Level
 95 						_replacement = _logEvent.getLevel().getName();
 96 						break;
 97 					case "r": // Milliseconds since startup
 98 						_replacement = "" + _logEvent.getTimeStamp().getDifference(jsx.logging.LogManager.START_TIMESTAMP);
 99 						break;
100 					case "%": // Literal % sign
101 						_replacement = "%";
102 						break;
103 					default:
104 						_replacement = _match;
105 						break;
106 			}
107 
108 			var len = 0;
109 			// truncation
110 			if(_truncation)
111 			{
112 				len = Number.parseInt(_truncation.substr(1));
113 				var strLen = _replacement.length;
114 				if(len < strLen)
115 				{
116 					_replacement = _replacement.substring(strLen - len, strLen);
117 				}
118 			}
119 
120 			// padding
121 			if(_padding)
122 			{
123 				if(_padding.charAt(0) == "-")
124 				{
125 					len = Number.parseInt(_padding.substr(1));
126 					// Right pad with spaces
127 					while (_replacement.length < len)
128 					{
129 						_replacement += " ";
130 					}
131 				}
132 				else
133 				{
134 					len = Number.parseInt(_padding);
135 					// Left pad with spaces
136 					while (_replacement.length < len)
137 					{
138 						_replacement = " " + _replacement;
139 					}
140 				}
141 			}
142 
143 			_msg.append(_replacement);
144 		}
145 
146 		return _msg.toString();
147 	}
