 60 	jsx.tunit.htmlgui.ResultPrinter.prototype.print = function(_test, _result, _runTime)
 61 	{
 62 		var me = this;
 63 
 64 		// title
 65 		var titleContainer = this.document.createElement("h3");
 66 		var title = this.document.createTextNode(_test.getName());
 67 		titleContainer.appendChild(title);
 68 		this.container.appendChild(titleContainer);
 69 
 70 		// summary of tests
 71 		this.container.appendChild(this.summaryTests);
 72 		this.container.appendChild(this.document.createElement("br"));
 73 
 74 		// summary text
 75         var summaryText = this.document.createTextNode(_result.runCount() + " tests ran in " + _runTime / 1000.0 + " seconds");
 76         this.container.appendChild(summaryText);
 77 		this.container.appendChild(this.document.createElement("br"));
 78 
 79 		// errors and failures
 80 		if(!_result.wasSuccessful())
 81 		{
 82 			if(_result.failureCount() > 0)
 83 			{
 84 				this.container.appendChild(this.document.createElement("br"));
 85 				var failureTitleContainer = this.document.createElement("div");
 86 				var failureTitle = this.document.createTextNode("Failures");
 87 				failureTitleContainer.appendChild(failureTitle);
 88 				failureTitleContainer.style.fontWeight = 'bold';
 89 				var failureContainer = this.document.createElement("div");
 90 				_result.getFailures().each(function(_failure)
 91 				{
 92 					failureContainer.appendChild(this.document.createTextNode(_failure.getTest() +" - "+ _failure.getException().message));
 93 					failureContainer.appendChild(this.document.createElement('br'));
 94 				});
 95 				failureContainer.style.color = 'red';
 96 				this.container.appendChild(failureTitleContainer);
 97 				this.container.appendChild(failureContainer);
 98 			}
 99 
100 			if(_result.errorCount() > 0)
101 			{
102 				this.container.appendChild(this.document.createElement("br"));
103 				var errorTitleContainer = this.document.createElement("div");
104 				var errorTitle = this.document.createTextNode("Errors");
105 				errorTitleContainer.appendChild(errorTitle);
106 				errorTitleContainer.style.fontWeight = 'bold';
107 				var errorContainer = this.document.createElement("div");
108 				_result.getErrors().each(function(_error)
109 				{
110 					errorContainer.appendChild(this.document.createTextNode(_error.getTest() +" - "));
111 
112 					errorContainer.appendChild(me.formatException(_error.getException()));
113 
114 					errorContainer.appendChild(this.document.createElement('br'));
115 				});
116 				errorContainer.style.color = 'red';
117 				this.container.appendChild(errorTitleContainer);
118 				this.container.appendChild(errorContainer);
119 			}
120 
121 			this.container.appendChild(this.document.createElement("br"));
122 		}
