362 	jsx.logging.WindowConsoleAppender.prototype.append = function(_logEvent)
363 	{
364 		// return if not showing the level
365 		if((!this.showTrace && jsx.logging.Level.TRACE.equals(_logEvent.getLevel())) ||
366 			(!this.showDebug && jsx.logging.Level.DEBUG.equals(_logEvent.getLevel())) ||
367 			(!this.showInfo && jsx.logging.Level.INFO.equals(_logEvent.getLevel())) ||
368 			(!this.showWarn && jsx.logging.Level.WARN.equals(_logEvent.getLevel())) ||
369 			(!this.showError && jsx.logging.Level.ERROR.equals(_logEvent.getLevel())) ||
370 			(!this.showFatal && jsx.logging.Level.FATAL.equals(_logEvent.getLevel())))
371 		{
372 			return;
373 		}
374 
375 		// log message as string
376 		var layout = this.getLayout();
377 		var msg = new jsx.lang.StringBuffer()
378 						.append(layout.getHeader())
379 						.append(layout.format(_logEvent))
380 						.append(layout.getFooter());
381 
382 		var msgColor = "";
383 		switch(_logEvent.getLevel().getNumber())
384 		{
385 			case jsx.logging.Level.TRACE_INT:
386 				msgColor = jsx.logging.WindowConsoleAppender.TRACE_COLOR;
387 				break;
388 			case jsx.logging.Level.DEBUG_INT:
389 				msgColor = jsx.logging.WindowConsoleAppender.DEBUG_COLOR;
390 				break;
391 			case jsx.logging.Level.INFO_INT:
392 				msgColor = jsx.logging.WindowConsoleAppender.INFO_COLOR;
393 				break;
394 			case jsx.logging.Level.WARN_INT:
395 				msgColor = jsx.logging.WindowConsoleAppender.WARN_COLOR;
396 				break;
397 			case jsx.logging.Level.ERROR_INT:
398 				msgColor = jsx.logging.WindowConsoleAppender.ERROR_COLOR;
399 				break;
400 			case jsx.logging.Level.FATAL_INT:
401 				msgColor = jsx.logging.WindowConsoleAppender.FATAL_COLOR;
402 				break;
403 			default:
404 				// nothing
405 		}
406 
407 		var msgCont = this.consoleWindow.createElement("div");
408 		msgCont.className = _logEvent.getLevel().getName();
409 
410 		var msgContStyle = msgCont.style;
411 		msgContStyle.height = "20px";
412 		msgContStyle.width = "95%";
413 		msgContStyle.borderBottom = "#ccc dashed 1px";
414 		msgContStyle.color = msgColor;
415 		msgContStyle.fontWeight = "bold";
416 		msgContStyle.fontFamily = "courier new, courier, monospace";
417 		msgContStyle.fontSize = "10px";
418 
419 		msgCont.appendChild(this.consoleWindow.createTextNode(msg));
420 
421 		this.loggingConsole.appendChild(msgCont);
422 
423 		msgCont.focus();
424 	}
