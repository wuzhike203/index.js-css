 97 	{
 98 		this.runTests += _test.countTestCases();
 99 
100 		this.listeners.each(function(_listener)
101 		{

103 		});
104 	}

114 	jsx.tunit.framework.TestResult.prototype.wasSuccessful = function()

119 	jsx.tunit.framework.TestResult.prototype.run = function(_test)
120 	{
121 		this.startTest(_test);
122 
123 		var _success = true;
124 		try
125 		{
126 			_test.runBare();
127 		}
128 		catch(_exception)
129 		{
130 			_success = false;
131 			if(jsx.tunit.framework.AssertionFailedError.klass.isInstance(_exception))
132 			{
133 				this.addFailure(_test, _exception);
134 			}
135 			else
136 			{
137 				this.addError(_test, _exception);
138 			}
139 		}
140 
141 		if(_success)
142 		{
143 			this.addSuccess(_test);
144 		}
145 
146 		this.endTest(_test);
147 	}
