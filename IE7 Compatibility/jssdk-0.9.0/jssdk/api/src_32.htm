 33 	jsx.event.Event.prototype.getEvent = function()

 36 	}

 38 	jsx.event.Event.prototype.getType = function()

 43 	jsx.event.Event.prototype.getObject = function()
 44 	{

 46 	}

 48 	jsx.event.Event.prototype.isControlKeyDown = function()
 49 	{

 51 	}

 53 	jsx.event.Event.prototype.isLeftClick = function()
 54 	{
 55 		return (((this.event.which) && (this.event.which == 1)) ||
 56 				((this.event.button) && (this.event.button == 1)));
 57 	}
 58 
 59 	jsx.event.Event.prototype.getPointerX = function()
 60 	{
 61 		return this.event.pageX ||
 62 				(this.event.clientX + (document.documentElement.scrollLeft || document.body.scrollLeft));
 63 	}
 64 
 65 	jsx.event.Event.prototype.getPointerY = function()
 66 	{
 67 		return this.event.pageY ||
 68 				(this.event.clientY + (document.documentElement.scrollTop || document.body.scrollTop));
 69 	}

 71 	jsx.event.Event.prototype.stopEvent = function()
 72 	{
 73 		if(this.event.stopPropagation) this.event.stopPropagation(); else this.event.cancelBubble = true;
 74 	}
 75 
 76 	jsx.event.Event.prototype.stopDefault = function()
 77 	{
 78 		if(this.event.preventDefault) this.event.preventDefault(); else this.event.returnValue = false;
 79 	}

 81 	jsx.event.Event.prototype.stopSelecting = function()
 82 	{
 83 		if(typeof document.onselectstart != 'undefined')
 84 		{
 85 			document.onselectstart = function(){return false;}
 86 		}
 87 	}

 89 	jsx.event.Event.prototype.startSelecting = function()
 90 	{
 91 		if(typeof document.onselectstart != 'undefined')
 92 		{
 93 			document.onselectstart = function(){return true;}
 94 		}
 95 	}
