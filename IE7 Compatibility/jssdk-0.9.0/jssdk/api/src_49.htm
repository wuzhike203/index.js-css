 15 	jsx.lang.StringBuffer.prototype.initialize = function(_initString)
 16 	{
 17 		this.str = null;
 18 		this.current = 0;
 19 		this.parts = [];

 21 
 22 		if(_initString != null)

 28 	/**
 29 	* Appends the _appendString String to the current buffer.
 30 	* @param {String} _appendString
 31 	* @return StringBuffer
 32 	*/
 33 	jsx.lang.StringBuffer.prototype.append = function(_appendString)
 34 	{
 35 		// append the string
 36 		this.parts[this.current++] = _appendString.toString();
 37 
 38 		// reset the add new lengh for appended string
 39 		this.length += _appendString.toString().length;
 40 
 41 		// reset cache
 42 		this.str = null;
 43 
 44 		// return the Instance for method chaining
 45 		return this;
 46 	}

 67 	/**
 68 	* Returns a String representation of a jsx.lang.StringBuffer.
 69 	* @return {String}
 70 	*/
 71 	jsx.lang.StringBuffer.prototype.toString = function()
 72 	{
 73 		if(this.str)
 74 		{
 75 			return this.str;
 76 		}
 77 
 78 		var newStr = this.parts.join("");
 79 		this.parts = [newStr];
 80 		this.current = 1;
 81 		this.length = newStr.length;
 82 		this.str = newStr;
 83 
 84 		return this.str;
 85 	}
