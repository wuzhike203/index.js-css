451 	* Returns the number of week in the year. Ex... 29.
452 	* @param {Number} _minimalDaysInFirstWeek
453 	* @return {Number}
454 	*/
455 	// This method was adopted from log4javascript -> http://www.timdown.co.uk/log4javascript
456 	Date.prototype.getWeekInYear = function(_minimalDaysInFirstWeek)
457 	{
458 		if(Object.isUndefined(_minimalDaysInFirstWeek)) {
459 			_minimalDaysInFirstWeek = Date.DEFAULT_MINIMAL_DAYS_IN_FIRST_WEEK;
460 		}
461 
462 		var _previousSunday = new Date(this.getTime() - (this.getDay() * Date.ONE_DAY_MILLISECONDS));
463 		_previousSunday = new Date(_previousSunday.getFullYear(), _previousSunday.getMonth(), _previousSunday.getDate());
464 
465 		var _startOfYear = new Date(this.getFullYear(), 0, 1);
466 
467 		var _numberOfSundays = _previousSunday.isBefore(_startOfYear) ?
468 								0 : 1 + Math.floor((_previousSunday.getTime() - _startOfYear.getTime()) / Date.ONE_WEEK_MILLISECONDS);
469 
470 		var _weekInYear = _numberOfSundays;
471 
472 		var _numberOfDaysInFirstWeek = 7 - _startOfYear.getDay();
473 		if(_numberOfDaysInFirstWeek >= _minimalDaysInFirstWeek)
474 		{
475 			_weekInYear++;
476 		}
477 
478 		return _weekInYear;
479 	}
