 19 	/**
 20 	* Append the specified object's toString keyed by the specified name.
 21 	* @param {Object} _object
 22 	* @param {String} _name
 23 	* @return {jsx.lang.util.ToStringBuilder}
 24 	*/
 25 	jsx.lang.util.ToStringBuilder.prototype.append = function(_object, _name)
 26 	{
 27 		this.str.append("\n");
 28 		this.str.append(_name);
 29 		this.str.append(" = ");
 30 		this.str.append(_object.toString());
 31 
 32 		return this;
 33 	}

 38 	*/

 42 	}
