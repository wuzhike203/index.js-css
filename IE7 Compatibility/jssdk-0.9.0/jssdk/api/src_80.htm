 17 	jsx.util.scheduler.Task.prototype.initialize = function(_block, _object, _args, _runInterval)
 18 	{
 19 		this.block = _block;
 20 		this.object = _object;
 21 		this.args = _args;
 22 
 23 		this.tickCount = Number.parseInt(_runInterval/jsx.util.scheduler.Scheduler.INTERVAL);
 24 		this.currentTick = this.tickCount;
 25 		this.running = false;
 26 	}

 34 	{

 36 	}

 38 	jsx.util.scheduler.Task.prototype.run = function()
 39 	{
 40 		this.currentTick--;
 41 		if (this.currentTick <= 0)
 42 		{
 43 			if(this.running)
 44 			{
 45 				new jsx.lang.Thread(this.block, this.object, this.args).start();
 46 			}
 47 
 48 			// reset the this.currentTick from the original this.tickCount
 49 			this.currentTick = this.tickCount;
 50 		}
 51 	}
 52 
 53 	jsx.util.scheduler.Task.prototype.isRunning = function()
 54 	{
 55 		return this.running;
 56 	}
