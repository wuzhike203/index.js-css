 50 	{
 51 		this.setUp();
 52 		try
 53 		{
 54 			this.runTest();
 55 		}
 56 		finally
 57 		{
 58 			this.tearDown();
 59 		}
 60 	}

 62 	jsx.tunit.framework.TestCase.prototype.runTest = function()
 63 	{
 64 		this.assertNotNull(this.name, "TestCase.name cannot be null");
 65 
 66 		var runMethod = this.getClass().getMethod(this.name);
 67 		if(runMethod)
 68 		{
 69 			runMethod.invoke(this);
 70 		}
 71 		else
 72 		{
 73 			this.fail("Method \""+this.name+"\" not found");
 74 		}
 75 	}
