457 	jsx.logging.ConsoleAppender.prototype.append = function(_logEvent)
458 	{
459 		// return if not showing the level
460 		if((!this.showTrace && jsx.logging.Level.TRACE.equals(_logEvent.getLevel())) ||
461 			(!this.showDebug && jsx.logging.Level.DEBUG.equals(_logEvent.getLevel())) ||
462 			(!this.showInfo && jsx.logging.Level.INFO.equals(_logEvent.getLevel())) ||
463 			(!this.showWarn && jsx.logging.Level.WARN.equals(_logEvent.getLevel())) ||
464 			(!this.showError && jsx.logging.Level.ERROR.equals(_logEvent.getLevel())) ||
465 			(!this.showFatal && jsx.logging.Level.FATAL.equals(_logEvent.getLevel())))
466 		{
467 			return;
468 		}
469 
470 		// log message as string
471 		var layout = this.getLayout();
472 		var msg = new jsx.lang.StringBuffer()
473 						.append(layout.getHeader())
474 						.append(layout.format(_logEvent))
475 						.append(layout.getFooter());
476 
477 		var msgColor = "";
478 		switch(_logEvent.getLevel().getNumber())
479 		{
480 			case jsx.logging.Level.TRACE_INT:
481 				msgColor = jsx.logging.ConsoleAppender.TRACE_COLOR;
482 				break;
483 			case jsx.logging.Level.DEBUG_INT:
484 				msgColor = jsx.logging.ConsoleAppender.DEBUG_COLOR;
485 				break;
486 			case jsx.logging.Level.INFO_INT:
487 				msgColor = jsx.logging.ConsoleAppender.INFO_COLOR;
488 				break;
489 			case jsx.logging.Level.WARN_INT:
490 				msgColor = jsx.logging.ConsoleAppender.WARN_COLOR;
491 				break;
492 			case jsx.logging.Level.ERROR_INT:
493 				msgColor = jsx.logging.ConsoleAppender.ERROR_COLOR;
494 				break;
495 			case jsx.logging.Level.FATAL_INT:
496 				msgColor = jsx.logging.ConsoleAppender.FATAL_COLOR;
497 				break;
498 			default:
499 				// nothing
500 		}
501 
502 		var winConsole = this.getConsole();
503 		var consoleDoc = winConsole.document;
504 
505 		var msgCont = consoleDoc.createElement("div");
506 		msgCont.className = _logEvent.getLevel().getName();
507 
508 		var msgContStyle = msgCont.style;
509 		msgContStyle.height = "20px";
510 		msgContStyle.width = "95%";
511 		msgContStyle.borderBottom = "#ccc dashed 1px";
512 		msgContStyle.color = msgColor;
513 		msgContStyle.fontWeight = "bold";
514 		msgContStyle.fontFamily = "courier new, courier, monospace";
515 
516 		msgCont.appendChild(consoleDoc.createTextNode(msg))
517 		consoleDoc.body.appendChild(msgCont);
518 
519 		this.scrollIndex += 20;
520 		if(!this.scrollLocked)
521 		{
522 			winConsole.scrollTo(0, this.scrollIndex);
523 		}
524 		else
525 		{
526 			// locked scrolling
527 			if(this.scrollIndexLocked == 0)
528 			{
529 				this.scrollIndexLocked = this.scrollIndex;
530 				winConsole.scrollTo(0, this.scrollIndexLocked);
531 			}
532 		}
533 	}
